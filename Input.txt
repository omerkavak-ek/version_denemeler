        class="form-control"
        placeholder="name@example.com"
    />
</div>

Input group
En üstte bulunan component'a ig-radius-left, en altta bulunan component'a ise ig-radius-right sınıfları verilmelidir.

<div class="input-group">
  <button class="btn btn-outline-secondary ig-radius-left" type="button" id="button-addon1">
    Button
  </button>
  <input
    type="text"
    class="form-control ig-radius-right"
    placeholder=""
    aria-label="Example text with button addon"
    aria-describedby="button-addon1"
  />
</div>
